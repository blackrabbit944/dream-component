# Message

1. 支持多种 icon 类型的消息通知
2. loading 消息可通过回调函数关闭

export const Basic = (args) => {
    return (
        <div style={{ margin: '0 12px', display: 'inline-block' }}>
            <Button onClick={args.onClick}>{args.children}</Button>
        </div>
    );
};

## 基础用法

```tsx
import { Button } from 'dream-component';

export const MyComponent = (args) => {
    return <Button />;
};
```
