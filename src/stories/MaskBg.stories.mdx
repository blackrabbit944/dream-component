# 覆盖层组件

## 基础用法

```tsx
import { MaskBg } from 'dream-components/maskbg';

export const MyComponent = (args) => {
    return <Mask visible={true} opacity={70} />;
};
```

export const Template = (args) => {
    let [visible, setVisible] = useState(false);
    args['visible'] = visible;
    console.log('args', args);
    return (
        <div>
            <div>这里是一些背景文字</div>
            <div>这里是一些背景文字</div>
            <MaskBg {...args} />
            {visible ? (
                <div
                    style={{
                        position: 'absolute',
                        bottom: '40%',
                        left: '50%'
                    }}
                >
                    <button
                        onClick={() => {
                            setVisible(false);
                            console.log('关闭弹窗');
                        }}
                    >
                        关闭弹窗
                    </button>
                </div>
            ) : (
                <div>
                    <button
                        onClick={() => {
                            setVisible(true);
                        }}
                    >
                        打开弹窗
                    </button>
                </div>
            )}
        </div>
    );
};
